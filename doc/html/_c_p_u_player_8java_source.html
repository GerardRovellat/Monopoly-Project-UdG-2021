163  private int optionalActionSelector(Player player,ArrayList<optionalActions> optional_actions) {
164  HashMap<Integer,Integer> bestMoves = new HashMap<>();
165  bestMoves.put(0,20);
166 
167  int pos_iterator = 1;
168  for (optionalActions opaction : optional_actions) {
169  switch (opaction.getClass().getName()) {
170  case "OpActBuy":
171  if (player.getMoney() >= 20000)
172  bestMoves.put(pos_iterator, player.getMoney() / 1000); // Ex: money = 40000 / 1000 = 40
173  else bestMoves.put(pos_iterator, 0);
174  break;
175  case "OpActSell":
176  if (player.getMoney() <= 20000 && player.getFields().size() > 0)
177  bestMoves.put(pos_iterator, player.getMoney() / 1000);
178  else bestMoves.put(pos_iterator, 0);
179  break;
180  case "OpActLoan":
181  if (player.getMoney() <= 5000)
182  bestMoves.put(pos_iterator, player.getMoney() / 1000 * 20); // Ex: money = 3000 => 3000/1000 = 3 => 5-3 = 2 => 2*20 = 40%
183  else bestMoves.put(pos_iterator, 0);
184  break;
185  case "OpActLuckCard":
186  List<Card> cards = player.getLuckCards();
187  int prob = 0;
188  for (Card aux : cards) {
189  switch (aux.getType()) {
190  case "CHARGE":
191  prob = prob + 40;
192  break;
193  case "GET":
194  prob = prob + 50;
195  break;
196  case "GO":
197  prob = prob + 10;
198  break;
199  }
200  }
201  bestMoves.put(pos_iterator, prob);
202  break;
203  }
204  pos_iterator++;
205  }
206  int bestMove = 0;
207  for (HashMap.Entry<Integer,Integer> entry : bestMoves.entrySet()) {
208  if (entry.getValue() > bestMoves.get(bestMove)) {
209  bestMove = entry.getKey();
210  }
211  }
212  return bestMove;
213  }
